Introduction
This is a slide library, with stacked slide styles.
Currently, the library supports two types, horizontal and vertical.
The library is still in the development, addition and update phase
Docs
import { StandSlide, HorizontalSlide } from 'overlapping_slide';
const slideData = [ { color: "#788aff", child: 1 }, { color: "#20898d", child: 2 }, { color: "#9bacc9", child: 3 }, { color: "#2f975e", child: 4 }, { color: "#4555ff", child: 5 }, ];
return( <StandSlide slideData = {slideData} spaceBetween={160}/> <StandSlide slideData = {slideData} spaceBetween={160} isAutoNext timeNext={5000}/> <HorizontalSlide slideData = {slideData} spaceBetween={107}/> <StandSlide slideData = {slideData} spaceBetween={160} isAutoNext timeNext={5000}/> )
